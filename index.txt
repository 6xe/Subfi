    <script>
        // بيانات المنتجات
        const products = [
            { name: 'اشتراك يوتيوب بريميوم', price: 5, image: '📺' },
            { name: 'اشتراك شات جي بي تي', price: 9.5, image: '🤖' },
            { name: 'بلس سوني 5 - سنة', price: 126, image: '🎮' }
        ];

        // متغيرات السلة
        let cart = [];
        let quantities = [1, 1, 1, 1];

        // تغيير الكمية
        function changeQuantity(productIndex, change) {
            quantities[productIndex] = Math.max(1, quantities[productIndex] + change);
            document.getElementById(`quantity-${productIndex}`).textContent = quantities[productIndex];
        }

        // إضافة منتج للسلة
        function addToCart(productIndex) {
            const product = products[productIndex];
            const quantity = quantities[productIndex];
            
            const existingItem = cart.find(item => item.productIndex === productIndex);
            if (existingItem) {
                existingItem.quantity += quantity;
            } else {
                cart.push({
                    productIndex,
                    name: product.name,
                    price: product.price,
                    image: product.image,
                    quantity
                });
            }
            
            updateCartDisplay();
            
            // أنيميشن النجاح
            const btn = event.target;
            const originalText = btn.textContent;
            btn.textContent = '✅ تمت الإضافة';
            btn.style.background = 'linear-gradient(45deg, #4CAF50, #45a049)';
            
            setTimeout(() => {
                btn.textContent = originalText;
                btn.style.background = 'linear-gradient(45deg, #667eea, #764ba2)';
            }, 1500);
        }

        // تحديث عرض السلة
        function updateCartDisplay() {
            const cartCount = cart.reduce((sum, item) => sum + item.quantity, 0);
            const cartTotal = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);
            
            document.getElementById('cart-count').textContent = cartCount;
            document.getElementById('cart-total').textContent = cartTotal + ' ر.س';
        }

        // فتح سلة التسوق
        function openCart() {
            const cartItemsContainer = document.getElementById('cart-items');
            const cartTotalDisplay = document.getElementById('cart-total-display');
            
            cartItemsContainer.innerHTML = '';
            
            if (cart.length === 0) {
                cartItemsContainer.innerHTML = '<p style="text-align: center; color: #666; padding: 20px;">السلة فارغة</p>';
                cartTotalDisplay.textContent = 'المجموع: 0 ر.س';
            } else {
                cart.forEach((item, index) => {
                    const cartItem = document.createElement('div');
                    cartItem.className = 'cart-item';
                    cartItem.innerHTML = `
                        <div>
                            <span style="font-size: 1.2rem; margin-left: 10px;">${item.image}</span>
                            <span style="font-weight: bold;">${item.name}</span>
                        </div>
                        <div>
                            <span>الكمية: ${item.quantity}</span>
                            <span style="margin: 0 10px;">×</span>
                            <span style="font-weight: bold;">${item.price} ر.س</span>
                            <button onclick="removeFromCart(${index})" style="
                                background: #ff4757; 
                                color: white; 
                                border: none; 
                                border-radius: 50%; 
                                width: 25px; 
                                height: 25px; 
                                margin-right: 10px; 
                                cursor: pointer;
                                font-size: 14px;
                            ">×</button>
                        </div>
                    `;
                    cartItemsContainer.appendChild(cartItem);
                });
                
                const total = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);
                cartTotalDisplay.textContent = `المجموع: ${total} ر.س`;
            }
            
            document.getElementById('cart-modal').style.display = 'block';
        }

        // حذف من السلة
        function removeFromCart(index) {
            cart.splice(index, 1);
            updateCartDisplay();
            openCart(); // إعادة فتح السلة لتحديث العرض
        }

        // إغلاق السلة
        function closeCart() {
            document.getElementById('cart-modal').style.display = 'none';
        }

        // إتمام الشراء
        function checkout() {
            if (cart.length === 0) {
                alert('السلة فارغة!');
                return;
            }
            
            const selectedPayment = document.querySelector('input[name="payment"]:checked').value;
            const paymentMethods = {
                'card': 'بطاقة ائتمانية',
                'paypal': 'PayPal',
                'cash': 'الدفع عند الاستلام',
                'bank': 'تحويل بنكي'
            };
            
            // إخفاء المودال
            closeCart();
            
            // عرض رسالة النجاح
            document.getElementById('success-message').style.display = 'block';
            
            // إخفاء رسالة النجاح بعد 3 ثواني
            setTimeout(() => {
                document.getElementById('success-message').style.display = 'none';
            }, 3000);
            
            // تفريغ السلة
            cart = [];
            updateCartDisplay();
            
            console.log('تم إتمام الطلب بطريقة الدفع:', paymentMethods[selectedPayment]);
        }

        // إغلاق المودال عند الضغط خارجه
        window.onclick = function(event) {
            const modal = document.getElementById('cart-modal');
            if (event.target === modal) {
                closeCart();
            }
        }

        // تحديث العرض الأولي
        updateCartDisplay();
    </script>
